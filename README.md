# tareasApp

#instalar ionic y cordova
-  npm install -g @ionic/cli cordova

#instalar los paquetes y correr localmente el proyecto
-  npm install
-  ionic serve

#generar compilado para publicacion en android apk e ios ipa
-  
-  ionic cordova platform add android
-  ionic cordova platform add ios
-  cordova build android
-  cordova build ios
